const RenderMembers = ({memberList}) =>{

  memberList.map((member) => {
    return <div className={'member flex flex-row justify-center sm:flex-col'}>
    <div className={'photo'}>

const RenderTeams = () =>{
return <div>
  <RenderMembers memberList={memberJson.leads} />
  {

      <div className={`team mt-4`}>
        <h2>{team.name}</h2>
        <div className={'member flex flex-row justify-center sm:flex-col'}>
          <div>
            <RenderMembers memberList={team.vp} />
          </div>

</div>
}
